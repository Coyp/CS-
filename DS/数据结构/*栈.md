### 计算器系列

- 224.基本计算器 (括号 + 加减, 非负数)

  ```c++
    for (int i = 0; i < s.size(); i++) {
        long num = 0;
        if (s[i] >= '0' && s[i] <= '9') {
            while (s[i] >= '0' && s[i] <= '9') {
                num = num * 10 + s[i] - '0';
                i++;
            }
            i--;
            sum += sign * num;
        }

        if (s[i] == '+') sign = 1;
        else if (s[i] == '-') sign = -1;
        else if (s[i] == '(') {
            st.push(sum);
            st.push(sign);
            sum = 0;
            sign = 1;
        } else if (s[i] == ')') {
            sum *= st.top(); st.pop();
            sum += st.top(); st.pop();
        }
    }
  ```

- 227.基本计算器II (仅四则运算, 非负数)
  - 都添上符号 2 - 12 + 2 * 3 / 2-> +2 - 12 + 2
  - 栈内放入值

  ```c++
  if (s[i] >= '0' && s[i] <= '9') {
    num = num * 10 + s[i] - '0';
  }
  if (s[i] < '0' && s[i] != ' ' || i == s.size() - 1) {
    if (sign == '+') st.push(num);
    else if (sign == '-') st.push(-num);
    else if (sign == '*' || sign == '/') {
      auto pre = st.top();
      st.pop();
      st.push(sign == '*' ? pre * num : pre / num);
    }
    sign = s[i];
    num = 0;
  }
  ```
