- 26.删除排序数组中的重复项

    ```go
    // 1 1 2 3 -> 1 2 3
    j := 1
    for i := 1; i < len(nums); i++ {
        if nums[i] != nums[i-1] {
            nums[j] = nums[i]
            j++
        }
    }
    ```

- 80.删除排序数组中的重复项 II

    ```go
    // 1 1 1 2 2 3 3 ->  1 1 2 2 3 3
    count := 0
    j := 1
    for i := 1; i < len(nums); i++ {
        if nums[i] != nums[i-1] {
            nums[j] = nums[i]
            j++
            count = 0
        } else if nums[i] == nums[i-1] && count < 1 { // 多了一个等于的条件
            nums[j] = nums[i]
            j++
            count++
        }
    }
    ```

- 283.移动零

    ```go
    int j = 0;
    for (int i = 0; i < nums.size(); i++) {
        if (nums[i] != 0) {
            swap(nums[i], nums[j]);
            j++;
        }
    }
    ```
